'use strict';

function hookIfExists(moduleName, symbolNames) {
    const mod = Process.findModuleByName(moduleName);
    if (!mod) {
        console.log('[!] módulo não carregado:', moduleName);
        return;
    }

    console.log('[+] módulo:', moduleName, 'base=', mod.base);

    symbolNames.forEach(name => {
        const addr = Module.findExportByName(moduleName, name);
        if (!addr) {
            console.log('  [-] símbolo não encontrado:', name);
            return;
        }

        console.log('  [+] hook em', name, '@', addr);

        Interceptor.attach(addr, {
            onEnter(args) {
                this.name = name;
                console.log('[CALL]', moduleName, name,
                    'arg0=', args[0],
                    'arg1=', args[1],
                    'arg2=', args[2],
                    'tid=', Process.getCurrentThreadId());
            },
            onLeave(retval) {
                console.log('[RET ]', moduleName, this.name, '=>', retval);
            }
        });
    });
}
